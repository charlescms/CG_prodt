select
      CONTRATOTRANSPORTE.CONTRATO    AS CONTRATOS,
      DIVISOES.NOME                   AS DIVISION ,
      FORNECEDORES.RAZAO             AS EXPORTADOR,
      CONTRATOTRANSPORTE.NOMEMIX     AS PRODUCTO
from  CONTRATOTRANSPORTE  , FORNECEDORES, DIVISOES
WHERE  CONTRATOTRANSPORTE.FORCOD = FORNECEDORES.FORCOD
 AND   CONTRATOTRANSPORTE.CLICOD = DIVISOES.CLICOD
 AND   CONTRATOTRANSPORTE.SEQ = DIVISOES.SEQ
