SELECT distinct
       ContratoTransporte.contrato  as "Contrato",
       Fornecedores.razao as "Provedor",
       ContratoTransporte.nomemix as "Produto",
       ROUND(ContratoTransporte.Comissao, 2)  || ' / '|| ContratoTransporte.FormComissao    as "% Comission" ,
       ContratoTransporte.FormComissao as "Unid.",
       ContratoTransporte.ValorTotalExp as "Valor Ejecutado",
       ExportadoTransito.Fatura as "No Factura",
       ContratoTransporte.QUANTTOTALEXP as "Cantidad",
       ExportadoTransito.ComisaoSFatura as "Valor de La Comision" ,
       ExportadoTransito.NumNFEmit as "No Nota Fiscal",
       ExportadoTransito.datanf as "Fecha de Emision NF",
       ExportadoTransito.Valor_NF as "Valor Fechamento"  ,
       ExportadoTransito.Recebimento  as   "Fecha de Cobro" 
 from  ContratoTransporte, ExportadoTransito,Fornecedores 
 where ContratoTransporte.contrato = ExportadoTransito.contrato 
   and ContratoTransporte.forcod =  Fornecedores.forcod 
 AND  CONTRATOTRANSPORTE.CADASTRO >= '2011/04/04' AND 
      CONTRATOTRANSPORTE.CADASTRO <= '2012/04/04' AND 
	  CONTRATOTRANSPORTE.FORCOD >=  1  AND 
	  CONTRATOTRANSPORTE.FORCOD <=  37
